int main()
{
    task("Word  Word   Word s ");
    return 0;
}
int task(char mas[]){
    int slova=0;
    for(int i=0;mas[i]!='\0';i++){
        if(mas[i]!=' '&&mas[i+1]==' '){
            slova++;
        }
    }
    printf("%d", slova);
}
