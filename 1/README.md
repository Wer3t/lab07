int task(int N, ...);
int main() {
	printf("%d ",task(5, 1, 2, 4, 3, 6));
	return 0;
}
int task(int N, ...){
	int rez = 0;
	int mas[N];
	va_list args;
	va_start(args, N);
	for (int i =0; i<N; i++){
		int value = va_arg(args, int);
		mas[i]=value;
	}
	for(int i =0; i<N; i++){
		if(i!=N-1){
			if(mas[i]<mas[i+1]){
				rez++;
			}
		}
	}
	va_end(args);
	return rez;
}
