int task(int N){
    int mas[N][N];
    int masr[N][N];
    for (int i=0;i<N;i++) {
    	for (int j=0;j<N;j++){
        	mas[i][j] = i + j;
        	masr[i][j] = 0;
        }
    }
    
    for (int i=0;i<N;i++) {
        for (int j=0;j<N;j++){
            for(int k=0; k<N; k++){
            	masr[i][j]+=mas[i][k]*mas[k][j];
            }     
            
       } 
    }
    for (int i=0;i<N;i++) {
    	for (int j=0;j<N;j++)
    	{
        	printf( "%d ", masr[i][j] );
    	}
    }
}
